// --- STATE VARIABLES ---

let currentLevel = 0;

let score = 0;

 

// --- FUNCTIONS ---

 

function startGame() {

currentLevel = 0;

score = 0;

updateScore();

document.getElementById('progressBar').style.width = '0%';

// Navigation

document.getElementById('introScreen').classList.add('hidden');

document.getElementById('victoryScreen').classList.add('hidden');

document.getElementById('gameScreen').classList.remove('hidden');

 

loadLevel();

}

 

function loadLevel() {

const levelData = levels[currentLevel];

// Set Text

document.getElementById('scenarioText').innerText = levelData.disaster;

 

// Generate Cards

const container = document.getElementById('optionsContainer');

container.innerHTML = ''; // Clear old cards

 

levelData.options.forEach(opt => {

const btn = document.createElement('div');

btn.className = 'card';

btn.onclick = () => checkAnswer(opt.id);

btn.innerHTML = `

<div class="card-icon"><i class="fa-solid ${opt.icon}"></i></div>

<div class="card-title">${opt.title}</div>

`;

container.appendChild(btn);

});

}

 

function checkAnswer(selectedId) {

const levelData = levels[currentLevel];

const overlay = document.getElementById('feedbackOverlay');

const title = document.getElementById('feedbackTitle');

const text = document.getElementById('feedbackText');

const nextBtn = document.querySelector('.next-btn');

 

overlay.classList.remove('hidden');

 

if (selectedId === levelData.correctId) {

// Correct

score++;

updateScore();

overlay.classList.remove('wrong');

overlay.classList.add('correct');

title.innerText = "SUPERPOWER ACTIVATED!";

title.style.color = "#10b981";

text.innerText = levelData.explanation;

nextBtn.innerText = "NEXT LEVEL";

} else {

// Wrong

overlay.classList.remove('correct');

overlay.classList.add('wrong');

title.innerText = "SYSTEM FAILURE!";

title.style.color = "#ef4444";

text.innerText = "That superpower didn't fix the problem. The server crashed!";

nextBtn.innerText = "TRY NEXT";

}

}

 

function nextLevel() {

document.getElementById('feedbackOverlay').classList.add('hidden');

currentLevel++;

// Update Progress Bar

const progress = (currentLevel / levels.length) * 100;

document.getElementById('progressBar').style.width = progress + '%';

 

if (currentLevel < levels.length) {

loadLevel();

} else {

showVictory();

}

}

 

function showVictory() {

document.getElementById('gameScreen').classList.add('hidden');

document.getElementById('victoryScreen').classList.remove('hidden');

document.getElementById('finalScore').innerText = score;

}

 

function updateScore() {

document.getElementById('scoreDisplay').innerText = `SCORE: ${score}`;

}
